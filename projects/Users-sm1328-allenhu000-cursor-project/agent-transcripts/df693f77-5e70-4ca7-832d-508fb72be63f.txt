/** 生成 6 位数字验证码 */
function randomCode(len = 6) {
  let s = '';
  for (let i = 0; i < len; i++) s += Math.floor(Math.random() * 10);
  return s;

/**
 * 向指定邮箱发送验证码邮件（需配置 SMTP 环境变量）
 * @param {string} to - 收件人邮箱
 * @param {Object} [options]
 * @param {string} [options.code] - 验证码，不传则自动生成 6 位数字
 * @param {string} [options.subject='您的验证码']
 * @param {string} [options.from] - 发件人，不传则用 SMTP_FROM
 * @param {Object} [options.smtp] - 覆盖 SMTP 配置，格式见下方
 * @returns {Promise<{ code: string, messageId?: string }>}
 *
 * 环境变量（可选，也可在 options.smtp 中传入）:
 *   SMTP_HOST, SMTP_PORT, SMTP_SECURE, SMTP_USER, SMTP_PASS, SMTP_FROM
 */
async function sendVerificationCode(to, options = {}) {
  const nodemailer = require('nodemailer');
  const code = options.code != null ? String(options.code) : randomCode(6);
  const subject = options.subject != null ? options.subject : '您的验证码';
  const smtp = options.smtp || {
    host: process.env.SMTP_HOST,
    port: process.env.SMTP_PORT ? parseInt(process.env.SMTP_PORT, 10) : 465,
    secure: process.env.SMTP_SECURE !== 'false',
    auth: process.env.SMTP_USER
      ? { user: process.env.SMTP_USER, pass: process.env.SMTP_PASS }
      : undefined,
  };
  const from = options.from || process.env.SMTP_FROM || smtp.auth?.user || 'noreply@local';

  const transporter = nodemailer.createTransport(smtp);
  const info = await transporter.sendMail({
    from,
    to,
    subject,
    text: `您的验证码是：${code}，请勿泄露。`,
    html: `<p>您的验证码是：<strong>${code}</strong>，请勿泄露。</p>`,
  });
  return { code, messageId: info.messageId };
}

## QQ 邮箱 SMTP 配置

### 1. 在 QQ 邮箱里开启服务并获取授权码

1. 打开 **https://mail.qq.com** 登录 QQ 邮箱  
2. 右上角 **设置** → **账户**  
3. 找到 **「POP3/IMAP/SMTP/Exchange/CardDAV/CalDAV 服务」**  
4. 开启 **「IMAP/SMTP 服务」** 或 **「POP3/SMTP 服务」**（开一个即可）  
5. 按提示用**绑定手机**发短信验证，验证后会显示 **授权码**（约 16 位）  
   - **重要**：这里用的是**授权码**，不是 QQ 登录密码  

### 2. 在项目里配置

1. 在项目根目录把 `.env.example` 复制成 `.env`  
2. 在 `.env` 里填写（其他项保持示例里的 QQ 配置即可）：

```env
SMTP_HOST=smtp.qq.com
SMTP_PORT=465
SMTP_SECURE=true
SMTP_USER=你的QQ号@qq.com
SMTP_PASS=刚才获取的16位授权码
SMTP_FROM=你的QQ号@qq.com
```

例如 QQ 号是 `123456789`，则：

- `SMTP_USER=123456789@qq.com`
- `SMTP_FROM=123456789@qq.com`
- `SMTP_PASS=` 填你在邮箱里拿到的那串授权码  

### 3. 测试

在项目根目录执行：

```bash
npm run email:send
```

若配置正确，会先创建临时邮箱，再给该邮箱发一封验证码邮件，然后轮询并打印收到的验证码。
