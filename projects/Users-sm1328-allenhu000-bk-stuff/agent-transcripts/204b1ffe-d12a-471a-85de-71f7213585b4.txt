  contents: /**
 * 金额输入格式化：按 DECIMAL_PLACES 限制小数位、按 MAX_AMOUNT_TIER_1 限制上限。
 * 与 onTemporaryPriceChange 等价格输入逻辑共用同一套规则。
 *
 * @param {string} value - 用户输入的原始字符串
 * @param {Object} [options]
 * @param {number} [options.decimalPlaces] - 小数位数，不传则从 ConfigCenter system.DECIMAL_PLACES 读取，默认 2
 * @param {number} [options.maxAmount] - 最大金额，不传则从 ConfigCenter goods.MAX_AMOUNT_TIER_1 读取
 * @param {boolean} [options.shouldToastOnExceed] - 超过上限时是否 toast，默认 true
 * @returns {{ value: string, exceeded: boolean }} 格式化后的字符串，以及是否因超限被截断
 */
function formatMoneyInput(value, options = {}) {
    const ConfigCenter = max.globalFn.ConfigCenter();
    const decimalPlaces = options.decimalPlaces ?? ConfigCenter.get('system', 'DECIMAL_PLACES') ?? 2;
    const maxAmount = options.maxAmount ?? ConfigCenter.get('goods', 'MAX_AMOUNT_TIER_1');
    const shouldToastOnExceed = options.shouldToastOnExceed !== false;

    if (decimalPlaces > 0) {
        value = value.replace(/[^\d.]/g, '');
        value = value.replace(/^\./g, '');
        value = value.replace(/\.{2,}/g, '.');
        const decimalRegex = new RegExp(`^(\\d+)\\.(\\d{${decimalPlaces}}).*$`);
        value = value.replace(decimalRegex, '$1.$2');
    } else {
        value = value.replace(/[^\d]/g, '');
    }

    let exceeded = false;
    if (maxAmount != null && parseFloat(value) > maxAmount) {
        value = String(maxAmount);
        exceeded = true;
        if (shouldToastOnExceed) {
            max.ui.toast(max.i18n.getText('70215e7ceb102a9e5bbf85e6800de8a5cbd5c257'));
        }
    }

    return { value, exceeded };
}
