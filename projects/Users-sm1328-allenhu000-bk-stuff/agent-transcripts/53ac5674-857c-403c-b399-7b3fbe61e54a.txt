访问站点时出现 **502 Bad Gateway**，一般是“网关/代理收到了无效或错误的来自上游服务器的响应”。常见原因包括：

## 1. 上游服务不可用或崩溃
- 后端应用进程挂了（例如 PHP-FPM、Node、Java 等）
- 服务器重启、部署、宕机
- 应用因未捕获异常而退出

## 2. 上游响应超时
- 后端处理太慢，超过 Nginx/负载均衡器的超时时间
- 数据库慢查询、外部 API 慢、死锁等导致请求卡住

## 3. 上游配置错误
- 反向代理（Nginx、Apache 等）里配置的上游地址/端口错误
- 上游服务换了端口或没监听预期端口
- 上游只监听 `127.0.0.1`，而代理从别的机器访问

## 4. 资源耗尽
- 服务器内存、CPU 用满，进程被系统杀掉（OOM）
- 连接数、文件描述符等达到上限，无法建立新连接

## 5. 防火墙/网络
- 防火墙或安全组拦截了网关到上游的流量
- 网关和上游不在同一网络或路由有问题

## 6. 应用层错误
- 后端返回了代理无法识别的响应（例如非 HTTP 格式、畸形响应）
- 上游返回 5xx，某些代理也会以 502 形式转发给用户

const BASE = 'https://api.mail.tm';

/**
 * 获取可用域名列表
 * @returns {Promise<{ domain: string }[]>}
 */
async function getDomains() {
  const res = await fetch(`${BASE}/domains`);
  if (!res.ok) throw new Error(`getDomains failed: ${res.status}`);
  const data = await res.json();
  const list = data['hydra:member'] || [];
  return list.filter((d) => d.isActive !== false).map((d) => ({ id: d.id, domain: d.domain }));
}

/**
 * 创建临时邮箱账号（自定义用户名）
 * @param {string} username - 邮箱前缀，如 'asiantest'（仅字母数字、点、下划线、连字符，建议小写）
 * @param {string} [password] - 密码，不传则自动生成
 * @returns {Promise<{ address: string, password: string, id: string }>}
 */
async function createInbox(username, password = randomString(12)) {
  const domains = await getDomains();
  if (!domains.length) throw new Error('No mail.tm domains available');
  const domain = domains[0].domain;
  const address = `${String(username).trim().toLowerCase()}@${domain}`;

  const res = await fetch(`${BASE}/accounts`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ address, password }),
  });
  const body = await res.json().catch(() => ({}));
  if (!res.ok) {
    const msg = body.message || body['hydra:description'] || res.statusText;
    throw new Error(`createInbox failed: ${res.status} - ${msg}`);
  }
  return { address, password, id: body.id };
}

/**
 * 获取 Bearer Token（创建账号后必须调用才能拉取邮件）
 * @param {string} address - 邮箱地址
 * @param {string} password - 密码
 * @returns {Promise<string>} token
 */
async function getToken(address, password) {
  const res = await fetch(`${BASE}/token`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ address, password }),
  });
  const data = await res.json().catch(() => ({}));
  if (!res.ok) throw new Error(`getToken failed: ${res.status} - ${data.message || res.statusText}`);
  if (!data.token) throw new Error('getToken: no token in response');
  return data.token;
}

/**
 * 获取收件箱邮件列表（摘要）
 * @param {string} token - Bearer token
 * @param {number} [page=1]
 * @returns {Promise<{ id, from, subject, intro, createdAt }[]>}
 */
async function getMessages(token, page = 1) {
  const res = await fetch(`${BASE}/messages?page=${page}`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  if (!res.ok) throw new Error(`getMessages failed: ${res.status}`);
  const data = await res.json();
  const list = data['hydra:member'] || [];
  return list.map((m) => ({
    id: m.id,
    from: m.from,
    subject: m.subject,
    intro: m.intro,
    createdAt: m.createdAt,
  }));
}

/**
 * 获取单封邮件完整内容（含 text/html，用于提取验证码）
 * @param {string} token
 * @param {string} messageId
 * @returns {Promise<{ subject, text, html, from, createdAt }>}
 */
async function getMessageById(token, messageId) {
  const res = await fetch(`${BASE}/messages/${messageId}`, {
    headers: { Authorization: `Bearer ${token}` },
  });
  if (!res.ok) throw new Error(`getMessage failed: ${res.status}`);
  const m = await res.json();
  return {
    subject: m.subject,
    text: m.text || '',
    html: (Array.isArray(m.html) ? m.html.join('') : m.html) || '',
    from: m.from,
    createdAt: m.createdAt,
  };
}

/**
 * 从文本中提取验证码（常见 4–8 位数字）
 * @param {string} text
 * @param {RegExp} [pattern] - 默认匹配 4–8 位数字
 * @returns {string|null}
 */
function extractVerificationCode(text, pattern = /\b(\d{4,8})\b/) {
  if (!text || typeof text !== 'string') return null;
  const m = text.match(pattern);
  return m ? m[1] : null;
}

/**
 * 轮询收件箱直到收到匹配的邮件并提取验证码
 * @param {string} token
 * @param {Object} [options]
 * @param {number} [options.pollIntervalMs=3000]
 * @param {number} [options.timeoutMs=120000]
 * @param {RegExp|string} [options.subjectFilter] - 主题包含或匹配
 * @param {RegExp} [options.codePattern=/\b(\d{4,8})\b/]
 * @returns {Promise<{ code: string, message: { subject, text, from } }>}
 */
async function waitForVerificationCode(token, options = {}) {
  const {
    pollIntervalMs = 3000,
    timeoutMs = 120000,
    subjectFilter = null,
    codePattern = /\b(\d{4,8})\b/,
  } = options;

  const subjectMatch = (subject) => {
    if (!subjectFilter) return true;
    if (typeof subjectFilter === 'string') return subject.includes(subjectFilter);
    return subjectFilter.test(subject);
  };

  const start = Date.now();
  let lastCount = 0;

  while (Date.now() - start < timeoutMs) {
    const list = await getMessages(token);
    if (list.length > lastCount) {
      for (const msg of list) {
        if (!subjectMatch(msg.subject)) continue;
        const full = await getMessageById(token, msg.id);
        const text = full.text || full.html.replace(/<[^>]+>/g, ' ');
        const code = extractVerificationCode(text, codePattern);
        if (code) {
          return {
            code,
            message: {
              subject: full.subject,
              text: full.text,
              from: full.from,
            },
          };
        }
      }
      lastCount = list.length;
    }
    await sleep(pollIntervalMs);
  }

  throw new Error('waitForVerificationCode: timeout, no matching code received');
}

## 三、API 速查

| 方法 | 说明 |
|------|------|
| `createInboxAndGetToken(username, password?)` | **推荐**。创建邮箱 + 取 Token，一步到位。`password` 不传则自动生成。 |
| `createInbox(username, password?)` | 仅创建邮箱，返回 `{ address, password, id }`。 |
| `getToken(address, password)` | 根据邮箱和密码获取 Bearer Token。 |
| `getMessages(token, page?)` | 获取收件列表（摘要：id、subject、from、intro、createdAt）。 |
| `getMessageById(token, messageId)` | 获取单封邮件完整内容（含 text、html）。 |
| `extractVerificationCode(text, pattern?)` | 从一段文本里用正则提取验证码，默认 `/\b(\d{4,8})\b/`。 |
| `waitForVerificationCode(token, options?)` | 轮询收件箱，直到某封邮件里能提取出验证码，返回 `{ code, message }`。 |
| `getDomains()` | 获取当前可用域名列表（一般用不到，创建邮箱时会自动选一个）。 |

### waitForVerificationCode 可选参数

| 参数 | 类型 | 默认 | 说明 |
|------|------|------|------|
| `pollIntervalMs` | number | 3000 | 轮询间隔（毫秒） |
| `timeoutMs` | number | 120000 | 超时时间（毫秒） |
| `subjectFilter` | string \| RegExp | - | 只处理主题包含该字符串或匹配该正则的邮件 |
| `codePattern` | RegExp | `/\b(\d{4,8})\b/` | 从正文中提取验证码的正则（第一个捕获组为验证码） |
